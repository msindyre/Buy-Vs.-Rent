<script>
let wealthChart=null;
function fmt(n){return '$'+Math.round(n).toLocaleString()}

function calculate(){
  const price=parseFloat(document.getElementById('price').value)||400000;
  const dpPct=parseFloat(document.getElementById('dp').value)||5;
  const rate=parseFloat(document.getElementById('rate').value)||6.8;
  const termYrs=parseInt(document.getElementById('term').value)||30;
  const taxPct=parseFloat(document.getElementById('tax').value)||1.1;
  const appre=parseFloat(document.getElementById('appre').value)||3;
  const hoa=parseFloat(document.getElementById('hoa').value)||0;
  const ins=parseFloat(document.getElementById('ins').value)||150;
  const maintPct=parseFloat(document.getElementById('maint').value)||1;
  const closePct=parseFloat(document.getElementById('close').value)||3;
  const pmiRate=parseFloat(document.getElementById('pmi').value)||0.6;
  const rent=parseFloat(document.getElementById('rent').value)||2000;
  const rentInc=parseFloat(document.getElementById('rentinc').value)||3;
  const rins=parseFloat(document.getElementById('rins').value)||20;
  const years=parseInt(document.getElementById('years').value)||10;
  const investRet=parseFloat(document.getElementById('invest').value)||7;

  const dp=price*dpPct/100;
  const loanAmt=price-dp;
  const closingCosts=price*closePct/100;
  const upfront=dp+closingCosts;
  const monthlyRate=rate/100/12;
  const n=termYrs*12;

  const pmt=monthlyRate>0?loanAmt*monthlyRate*Math.pow(1+monthlyRate,n)/(Math.pow(1+monthlyRate,n)-1):loanAmt/n;
  const monthlyTax=price*taxPct/100/12;
  const monthlyMaint=price*maintPct/100/12;
  const pmiMo=dpPct<20?loanAmt*pmiRate/100/12:0;
  const totalBuyMo=pmt+monthlyTax+ins+hoa+monthlyMaint+pmiMo;

  document.getElementById('buy-breakdown').innerHTML=
    row('Principal & interest',fmt(pmt))+
    row('Property taxes',fmt(monthlyTax))+
    row('Home insurance',fmt(ins))+
    (hoa>0?row('HOA fees',fmt(hoa)):'')+
    row('Maintenance (est.)',fmt(monthlyMaint))+
    (pmiMo>0?row('PMI',fmt(pmiMo)):'')+
    totalRow('Total monthly cost',fmt(totalBuyMo));

  document.getElementById('rent-breakdown').innerHTML=
    row('Monthly rent',fmt(rent))+
    row("Renter's insurance",fmt(rins))+
    row('Maintenance','$0')+
    row('Building equity','$0')+
    totalRow('Total monthly cost',fmt(rent+rins));

  const buyNWArr=[],rentNWArr=[],equityArr=[],yLabels=[],tableRows=[];

  for(let y=1;y<=years;y++){
    const homeVal=price*Math.pow(1+appre/100,y);
    let bal=loanAmt;
    const mr=monthlyRate;
    for(let m=0;m<y*12;m++){const i=bal*mr;bal-=(pmt-i);if(bal<0)bal=0;}
    const equity=homeVal-bal;
    const rentNWInvested=upfront*Math.pow(1+investRet/100,y);
    let diffInvested=0,cr=rent;
    for(let yi=0;yi<y;yi++){
      const diff=totalBuyMo*12-cr*12;
      if(diff>0)diffInvested=(diffInvested+diff)*(1+investRet/100);
      cr*=(1+rentInc/100);
    }
    const buyNW=Math.round(equity);
    const rentNW=Math.round(rentNWInvested-diffInvested);
    buyNWArr.push(buyNW);rentNWArr.push(rentNW);equityArr.push(Math.round(equity));yLabels.push('Yr '+y);
    tableRows.push({y,homeVal:Math.round(homeVal),loanBal:Math.round(bal),equity:Math.round(equity),buyNW,rentNW,gap:buyNW-rentNW});
  }

  const finalGap=buyNWArr[years-1]-rentNWArr[years-1];
  const vBox=document.getElementById('verdict-box');
  const vHead=document.getElementById('verdict-head');
  const vDetail=document.getElementById('verdict-detail');

  if(finalGap>5000){
    vBox.className='verdict buy';
    vHead.textContent='Buying builds significantly more wealth over '+years+' years';
    vDetail.textContent='After '+years+' years, buying puts you '+fmt(finalGap)+' ahead. Your home equity of '+fmt(equityArr[years-1])+' on a property worth '+fmt(tableRows[years-1].homeVal)+' is your primary wealth driver — even after accounting for the opportunity cost of your down payment.';
  } else if(finalGap<-5000){
    vBox.className='verdict rent';
    vHead.textContent='Consider a longer horizon — renting may lead over '+years+' years';
    vDetail.textContent='With these inputs, investing the down payment and difference leaves you '+fmt(Math.abs(finalGap))+' ahead over '+years+' years. Try extending the time horizon — buying typically overtakes renting between 7–10 years.';
  } else {
    vBox.className='verdict neutral';
    vHead.textContent='Buying and renting are financially close over '+years+' years';
    vDetail.textContent='The wealth difference after '+years+' years is '+fmt(Math.abs(finalGap))+'. At this level, lifestyle factors — stability, flexibility, community — may carry more weight than pure financial return.';
  }

  document.getElementById('wealth-cards').innerHTML=
    metricCard('Buy: net worth at yr '+years,fmt(buyNWArr[years-1]),'Home equity primarily',finalGap>=0?'gold':'muted')+
    metricCard('Rent: net worth at yr '+years,fmt(rentNWArr[years-1]),'Invested savings','muted')+
    metricCard('Wealth gap (buy advantage)',fmt(finalGap),'How much more buying builds',finalGap>=0?'gold':'rust')+
    metricCard('Home value at yr '+years,fmt(tableRows[years-1].homeVal),'At '+appre+'% annual appreciation','teal');

  const tbody=document.getElementById('yr-tbody');
  tbody.innerHTML='';
  tableRows.forEach(r=>{
    const isBreakeven=r.gap>0&&(r.y===1||tableRows[r.y-2].gap<=0);
    const hl=isBreakeven?' class="highlight"':'';
    const gc=r.gap>=0?'pos':'neg';
    const gp=r.gap>=0?'+':'';
    tbody.innerHTML+='<tr'+hl+'><td>Year '+r.y+(isBreakeven?' &#9733;':'')+'</td><td>'+fmt(r.homeVal)+'</td><td>'+fmt(r.loanBal)+'</td><td>'+fmt(r.equity)+'</td><td>'+fmt(r.buyNW)+'</td><td>'+fmt(r.rentNW)+'</td><td class="'+gc+'">'+gp+fmt(r.gap)+'</td></tr>';
  });

  if(wealthChart)wealthChart.destroy();
  const ctx=document.getElementById('wealthChart').getContext('2d');
  wealthChart=new Chart(ctx,{
    type:'bar',
    data:{
      labels:yLabels,
      datasets:[
        {type:'bar',label:'Home equity',data:equityArr,backgroundColor:'rgba(174,199,201,0.55)',borderColor:'#aec7c9',borderWidth:1,order:2},
        {type:'line',label:'Buy net worth',data:buyNWArr,borderColor:'#b8976a',backgroundColor:'rgba(184,151,106,0.08)',borderWidth:2,pointRadius:3,pointBackgroundColor:'#b8976a',fill:false,order:1},
        {type:'line',label:'Rent net worth',data:rentNWArr,borderColor:'#a05840',backgroundColor:'rgba(160,88,64,0.05)',borderWidth:2,borderDash:[5,4],pointRadius:3,pointBackgroundColor:'#a05840',fill:false,order:0}
      ]
    },
    options:{
      responsive:true,maintainAspectRatio:false,
      plugins:{
        legend:{display:false},
        tooltip:{callbacks:{label:ctx=>ctx.dataset.label+': '+fmt(ctx.parsed.y)}}
      },
      scales:{
        x:{ticks:{font:{size:11},color:'#888'},grid:{display:false}},
        y:{ticks:{callback:v=>'$'+Math.round(v/1000)+'k',font:{size:11},color:'#888'},grid:{color:'rgba(0,0,0,0.05)'}}
      }
    }
  });

  document.getElementById('results').classList.add('visible');
  document.getElementById('results').scrollIntoView({behavior:'smooth',block:'start'});
}

function totalRow(l,a){return '<div class="breakdown-row total"><span class="item">'+l+'</span><span class="amount">'+a+'</span></div>'}
function metricCard(label,value,sub,color){
  const colors={gold:'#8a6e47',teal:'#4a8385',rust:'#a05840',muted:'#6b6459'};
  return '<div class="metric-card"><div class="metric-label">'+label+'</div><div class="metric-value" style="color:'+colors[color]+'">'+value+'</div><div class="metric-sub">'+sub+'</div></div>';
}
